Сравнивать массивы можно, проверяя их равенство или неравенство либо экви-валентность или неэквивалентность. Равенство массивов – это когда совпадают все пары ключ / значение элементов массивов. Эквивалентность – когда кроме ра-венства значений и ключей элементов требуется еще, чтобы элементы в обоих массивах были записаны в одном и том же порядке. Равенство значений в PHP обозначается символом " == ", а эквивалентность – символом " === ".
<?
$a = array("и"=>"Информатика", "м"=>"Математика");
$b = array("м"=>"Математика", "и"=>"Информатика");
if ($a == $b) echo "Массивы равны и";
  else echo "Массивы НЕ равны и ";
if ($a === $b) echo " эквивалентны";
  else echo " НЕ эквивалентны";
// получим echo "Массивы равны и НЕ эквивалентны"
?>
